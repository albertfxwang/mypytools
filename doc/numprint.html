Utilities for formatting and printing one-dimensional Numeric arrays.
Usage example:
  
# >>> from Numeric import *
# >>> from numprint import *
>>> x = arange(5.)
>>> y = x*2.
>>> z = sqrt(x)
>>> l = format("%10.1f %10.2f",x,y)
>>> l.heading("%10s %10s" % ("x","2x"))
>>> print l
         x         2x
       0.0       0.00
       1.0       2.00
       2.0       4.00
       3.0       6.00
       4.0       8.00
 
>>> l.addcols("%10.3f",z)
>>> l.addheading(" %10s" % "sqrt(x)")
>>> print l
         x         2x    sqrt(x)
       0.0       0.00      0.000
       1.0       2.00      1.000
       2.0       4.00      1.414
       3.0       6.00      1.732
       4.0       8.00      2.000
